Problem:  My backend does not trust my frontend (on localhost:4207) / showing a CORS error
------------------------------------------------------------------------------------------
Problem:  I run the Angular webapp in split front-end & backend mode
		  Backend is listening on localhost:8080/app1
		  Frontend is listening on localhost:4207
		  -- But, frontend shows CORS error

Solution: Tell Spring to allow requests from localhost:4207


@Configuration
public class WebMvcConfig implements WebMvcConfigurer {

    @Value("${disable.cors}")
    private boolean disableCors;


    /**
     * Allow CORS requests to come from anywhere
     * -- Should be used for local debugging only
     * @param registry holds the CorsRegistry
     */
    @Override
    public void addCorsMappings(CorsRegistry registry) {
        if (disableCors) {
            // Loosen the restrictions so local developers can upload files from http://localhost:42007 to the backend
            registry.addMapping("/**")
                    .allowedOrigins("http://localhost:4200", "http://localhost:4201",  "http://localhost:4202",  "http://localhost:4203",  "http://localhost:4204")
                    .allowCredentials(true)
                    .allowedMethods("GET", "POST", "PUT", "DELETE");
        }
    }
}
