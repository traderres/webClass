Sample Batch Script to Get the Latest Development Branch for Lots of Projects
-----------------------------------------------------------------------------

@ECHO OFF
cd /d %APP_PROJECT_PATH%\core-library
IF %ERRORLEVEL% neq 0 goto Compile_failed

REM ** Add the tee program to your PATH **
REM ** To find it, run these commands
REM ** terminal> cd \
REM ** terminal> dir /s tee.*
REM **
SET PATH=c:\users\%USERNAME%\AppData\Local\Programs\Git\usr\bin;C:\Program Files\Git\usr\bin;%PATH%



ECHO.
ECHO ----------------------------------------------------------------
ECHO Getting latest development for core-library
cd  ..\core-library
IF %ERRORLEVEL% neq 0 goto Compile_failed
git checkout -b stuff
git branch -D development
CALL :GIT_FETCH_LOOP
git checkout -f development
git branch -D stuff
git status

ECHO.
ECHO Compiling core-library
CALL mvn clean install | tee c:\tools\scripts\maven-output.txt

REM Look for the "BUILD SUCCESS" string in the output file
findstr /n /c:"BUILD SUCCESS"  c:\tools\scripts\maven-output.txt

IF %ERRORLEVEL% neq 0 goto Compile_failed



ECHO.
ECHO ----------------------------------------------------------------
ECHO Getting latest development for foci-library
cd  ..\foci-library
IF %ERRORLEVEL% neq 0 goto Compile_failed
git checkout -b stuff
git branch -D development
CALL :GIT_FETCH_LOOP
git checkout -f development
git branch -D stuff
git status

ECHO.
ECHO Compiling foci-library
call mvn clean install | tee c:\tools\scripts\maven-output.txt

REM Look for the "BUILD SUCCESS" string in the output file
findstr /n /c:"BUILD SUCCESS"  c:\tools\scripts\maven-output.txt

IF %ERRORLEVEL% neq 0 goto Compile_failed


ECHO.
ECHO ----------------------------------------------------------------
ECHO Getting latest development for core-admin-webapp
cd  ..\core-admin-webapp
IF %ERRORLEVEL% neq 0 goto Compile_failed
git checkout -b stuff
git branch -D development
CALL :GIT_FETCH_LOOP
git checkout -f development
git branch -D stuff
git status


ECHO.
ECHO ----------------------------------------------------------------
ECHO Getting latest development for core-govt-registration-webapp
cd  ..\core-govt-registration-webapp
IF %ERRORLEVEL% neq 0 goto Compile_failed
git checkout -b stuff
git branch -D development
CALL :GIT_FETCH_LOOP
git checkout -f development
git branch -D stuff
git status


ECHO.
ECHO It Succeeded
GOTO End1


:Compile_failed
  ECHO.
  ECHO THERE WAS A PROBLEM.  Perhaps, you need to clone a project?  Stopping here.
GOTO End1



REM ** START OF GIT FETCH LOOP **
:GIT_FETCH_LOOP
echo Attempting to fetch from remote...

REM Run the git fetch command
git fetch -a

REM Check if the command was successful (ERRORLEVEL 0)
IF %ERRORLEVEL% EQU 0 (
    echo Fetch successful!
    goto :GIT_FETCH_LOOP_SUCCESS
)

echo Fetch failed. Retrying in 5 seconds...
c:\windows\system32\timeout /t 5 /nobreak >nul
goto :GIT_FETCH_LOOP

:GIT_FETCH_LOOP_SUCCESS
REM ** END OF GIT FETCH LOOP **


:TheRealEnd
