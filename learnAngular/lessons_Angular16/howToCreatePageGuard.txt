How to Create a Page Guard that Prevents Users from Opening Unauthorized Page Routes
------------------------------------------------------------------------------------
Problem:  Yes, I know the front-end is not secure
          But, I don't want users connecting to page routes that they don't have access to
Solution: Create a Page Guard
          Apply the Page Guard to all pages (except the forbidden and not-found pages)


Assumptions
 A) There is a front-end userService
 B) There is a userService.getLoggedInUserInfo() that returns a DTO that holds the allowed page routes


Procedure
---------
 1. Create the page guard

 2. Fill-it in

	import {ActivatedRouteSnapshot, CanActivateFn, Router, RouterStateSnapshot} from '@angular/router';
	import {inject, Injectable} from "@angular/core";
	import {map, Observable} from "rxjs";
	import {UserService} from "../services/user.service";
	import {UserInfoDTO} from "../models/user/user-info-dto";
	import {Constants} from "../utilities/constants";

	export namespace PageGuard {

		@Injectable({
			providedIn: 'root'
		})
		class InternalPageGuardService {

			constructor(private router: Router,
						private userService: UserService) {}


			/*
			 * Get the "configured" route
			 */
			private getConfiguredUrl(aActivatedRouteSnapshot: ActivatedRouteSnapshot): string {
				  return aActivatedRouteSnapshot.pathFromRoot
					.filter(v => v.routeConfig)
					.map(v => v.routeConfig!.path)
					.join('/');
			}


			public canActivate(aActivatedRouteSnapshot: ActivatedRouteSnapshot): Observable<boolean> {

				return this.userService.getLoggedInUserInfo().pipe(
					map((userInfoDTO: UserInfoDTO) => {
						// I got the UserInfoDTO from the back-end (or cache)

						// Build the requested page route (starting from the root)
						let nextUrl: string = this.getConfiguredUrl(aActivatedRouteSnapshot);

						if (!nextUrl) {
							return false;
						}

						// Check if the url is allowed
						let routeAllowed: boolean | undefined = userInfoDTO.allowedPageRoutes.get(nextUrl);

						if (!routeAllowed) {
							  this.router.navigate([Constants.FORBIDDEN_ROUTE]).then();

							  // Return false so that the router will *NOT* proceed with the route
							  // -- Instead the user is routed to the FORBIDDEN page
							  return false;
						  }

						  // Return true so the router will allow the route change to proceed.
						  return true;
					}));

			} // end of canActivate()

		}  // end of InternalPageGuardService


		/*
		 * Problem:    Angular is deprecating the CanActivate interface
		 * Solution:   if you have older guards, then you can still use them
		 *             Convert your older guard into a service and inject it below
		 */
		export const canActivate: CanActivateFn = (aActivatedRouteSnapshot: ActivatedRouteSnapshot, aRouterStateSnapshot: RouterStateSnapshot) => {
			// Inject the InternalPageGuardService and call canActivate()
			return inject(InternalPageGuardService).canActivate(aActivatedRouteSnapshot);
		};

	}


 3. Apply the page guard to your pages
    a. Edit your core-admin-routing.module.ts (or where ever your routes are defined)
    b. Apply the PageGuard


			// This module is LAZY LOADED
			// The prefix path of "admin/core" is set in the app-routing.module.ts
			const routes: Routes = [

			  // Every route below has the prefix "page/admin/core/ " added to the BEGINNING of the url
			  { path: 'allowed-file-types',         component: AllowedFileTypesGridComponent,     canActivate: [PageGuard.canActivate ] },    // map "page/admin/core/allowed-file-types"
			  { path: 'list-exceptions',            component: ListExceptionsGridComponent,       canActivate: [PageGuard.canActivate ] },    // map "page/admin/core/list-exceptions"
			  { path: 'manage-banners',             component: ManageBannersComponent,            canActivate: [PageGuard.canActivate ] },    // map "page/admin/core/manage-banners"
			  { path: 'manage-users',               component: ManageUsersComponent,              canActivate: [PageGuard.canActivate ] },    // map "page/admin/core/manage-users"
			];



			@NgModule({
			  imports: [RouterModule.forChild(routes)],
			  exports: [RouterModule]
			})
			export class CoreAdminRoutingModule { }
