How to Upgrade Highcharts v11.4.7 to v12.6.0
--------------------------------------------
Highcharts v12 had some breaking changes with the imports


How to import timeline module
-----------------------------
// Import the timeline module for Highcharts v11
import {Chart, Options} from "highcharts";
import * as Highcharts from 'highcharts';
import HC_timeline from 'highcharts/modules/timeline';
HC_timeline(Highcharts);


// Import the timeline module for highcharts v12
import {Chart, Options} from "highcharts";
import Highcharts from 'highcharts';
import 'highcharts/modules/timeline';



How to import accessibility & network graph module
--------------------------------------------------
// Import accessibility and network graph module in Highcharts v11
import * as Highcharts from 'highcharts';
import NetworkGraphModule from 'highcharts/modules/networkgraph';
import AccessibilityModule from 'highcharts/modules/accessibility';
NetworkGraphModule(Highcharts);
AccessibilityModule(Highcharts);


// Import accessibility and network graph module in Highcharts v12
import Highcharts from 'highcharts';
import 'highcharts/modules/networkgraph';
import 'highcharts/modules/accessibility';




How to Import the World Map
---------------------------
// importing the world map in Highcharts v11
import worldMap from '@highcharts/map-collection/custom/world.geo.json';
HighchartsMap;


// importing the world map in Highcharts v12
import worldMap from '@highcharts/map-collection/custom/world.topo.json';
HighchartsMap;




Tool Tip Changes for maps
-------------------------
    // In Highcharts v11, we needed a formatter metohd
    tooltip: {
      formatter: function () {
        return this.point.name; // Display only the country name
      },
    },


    // In Highcharts v12
    tooltip: {
      headerFormat: '',
      pointFormat: '{point.name}'
    },



Title String cannot be null
---------------------------
    // in Highcharts v11, we could have a null title
    yAxis: {
        labels: {
          enabled: false,
        },
        title: {
          text: null,
        },
      },


    // in Highcharts v12, we could have an empty string for the title
    yAxis: {
        labels: {
          enabled: false,
        },
        title: {
          text: '',
        },
      },
