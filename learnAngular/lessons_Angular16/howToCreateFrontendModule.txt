-----------------------------------------

Assumptions
1) You want to add front-end module:  "my-module"
2) You want to add this backend:      "my-module-backend"

11. Add this service:  UserService.java
    a. Create the services package
    b. Create a new java class:  UserService.java
            package gov.dcsa.myBackend.services;        // Change this to your package name

            import gov.dcsa.core.library.exceptions.MyRuntimeException;
            import gov.dcsa.core.library.interfaces.MyUserInfoInterface;
            import gov.dcsa.core.library.interfaces.UserServiceInterface;
            import gov.dcsa.core.library.security.MyDefaultOidcUser;
            import gov.dcsa.core.library.security.MyUserInfo;
            import org.springframework.beans.factory.annotation.Value;
            import org.springframework.security.core.Authentication;
            import org.springframework.security.core.context.SecurityContext;
            import org.springframework.security.core.context.SecurityContextHolder;
            import org.springframework.stereotype.Service;


            @Service
            public class UserService implements UserServiceInterface  {


                @Value("${security.mode}")
                private String securityMode;


                public MyUserInfo getUserInfo() {
                    if (this.securityMode.equalsIgnoreCase("keycloak")) {
                        return getUserInfoFromKeycloak();
                    }
                    else {
                        return getUserInfoFromNonKeyCloak();
                    }
                }


                public String getLoggedInUserName() {
                    MyUserInfo userinfo = getUserInfo();

                    return userinfo.getUsername();
                }

                public String getLoggedInFullName() {
                    MyUserInfo userinfo = getUserInfo();

                    return userinfo.getLoggedInFullName();
                }

                private MyUserInfo getUserInfoFromNonKeyCloak() {
                    // Get the UserInfo object from Spring Security
                    SecurityContext securityContext = SecurityContextHolder.getContext();
                    if (securityContext == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromNonKeyCloak():  SecurityContext is null.  This should never happen.");
                    }

                    Authentication auth = securityContext.getAuthentication();
                    if (auth == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromNonKeyCloak():  Authentication is null.  This should never happen.");
                    }

                    MyUserInfo userInfo = (MyUserInfo) auth.getPrincipal();
                    if (userInfo == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromNonKeyCloak():  UserInfo is null.  This should never happen.");
                    }

                    return userInfo;
                }


                private MyUserInfo getUserInfoFromKeycloak() {
                    // Get the UserInfo object from Spring Security
                    SecurityContext securityContext = SecurityContextHolder.getContext();
                    if (securityContext == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromKeycloak():  SecurityContext is null.  This should never happen.");
                    }

                    Authentication auth = securityContext.getAuthentication();
                    if (auth == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromKeycloak():  Authentication is null.  This should never happen.");
                    }

                    MyDefaultOidcUser myDefaultOidcUser = (MyDefaultOidcUser) auth.getPrincipal();
                    if (myDefaultOidcUser == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromKeycloak():  MyDefaultOidcUser is null.  This should never happen.");
                    }

                    MyUserInfo myUserInfo = myDefaultOidcUser.getMyUserInfo();
                    if (myUserInfo == null) {
                        throw new MyRuntimeException("Error in getUserInfoFromKeycloak():  MyUserInfo is null.  This should never happen.");
                    }

                    return myUserInfo;
                }

                public Integer getLoggedInUserId() {
                    MyUserInfo userinfo = getUserInfo();

                    return userinfo.getId();
                }


                /*
                 * This is used by the AuthenticationFailedInterceptor to get the MyUserInfo object
                 */
                @Override
                public MyUserInfoInterface getUserInfoFromWebApp() {
                    MyUserInfo myUserInfo = getUserInfo();

                    return myUserInfo;
                }
            }

Part 4 / Add the front-end module to the big-frontend-webapp
-------------------------------------------------------------

    cd %NI2_PROJECT_PATH%\big-frontend-webapp

    cd frontend
    ng g module modules/my --routing

                        |

    a. Add this file:  frontend/src/app/modules/my-module/environments/environment.development.ts

            export const environment = {

Part 5 / Add files needed to run the backend in Docker
------------------------------------------------------
 1. Create a deployment kubernetes yaml for your backend
    a. Open the infrastructure project
    b. Create a branch of infrastructure project
    c. COPY deploy-core-admin-backend.yaml --> PASTE --> Your new file
       The format:  deploy-<NAME>.yaml
    d. Search & Replace:
       Search for:    core-admin
       Replace with:  <new module name>

 2. Create a windows batch script that will delete, compile, and deploy the new backend:
    a. Go to the infrastructure/scripts/developer/windows
    b. COPY reloadCoreAdminBackend.bat --> PASTE --> Your new file
    c. Update the new batch script with your backend

 3. Create a unix shell script that will delete, compile, and deploy the new backend:
    a. Go to the infrastructure/scripts/developer/unix
    b. Copy reloadCoreAdminBackend.sh --> PASTE --> Your new file
    c. Update the new shell script with your backend

 4. Update the ni2-apps-ingress.yaml with the context path for the new backend
    a. Go to the infrastructure/kubernetes/ingress-ni2-apps.yaml
    b. Add this to the bottom:

          - path: /cab              <---------- This should match the name of your context path
            pathType: Prefix
            backend:
              service:
                name: core-admin-service   <---- This should be the name of your new backend
                port:
                  number: 443

    c. Reload the ingress
       cd %NI2_PROJECT_PATH%\infrastructure\kubernetes
       kubectl delete -f ingress-ni2-apps.yaml
       kubectl apply  -f ingress-ni2-apps.yaml

    d. Wait 60 seconds for Docker to finish loading the ingress

 5. Try it out
    nukeDatabaseAddRegisteredUsers          # To load the docker database with your changes to the core.applications table
    reloadCoreAdminBackend.bat
    -- Verify that it compiles and starts up

-----------------------------------------------

    b. Login as admin
       Password is admin
    c. Go to Realms -> MyRealm -> Clients                  !!! IMPORTANT STEP HERE !!!
    d. Add a client:
       In the Create client page
            Client type:    OpenID Connect
            Client ID:      core-admin-backend             # Change this to your new backend
            Client Name:    core-admin-backend             # Change this to your new backend
            Press "Next"
            Press "next"                                Change this to your context path
            Press "save"                                             |
       In the "Settings" tab                                         V
            Root URL:                   https://app.localhost.internal/cab
            Home URL:                   https://app.localhost.internal/cab
            Client authentication:      On
            Authorization:              On
            Authentication Flow:        Standard flow
            Login theme:                ni2-login-page-theme
            Press "Save"


 2. Export the realm settings to MyRealm-realm.json
    -- Followed the procedures in infrastructure/docs/howToUpdateKeycloakRealm.txt from step 4 on

 3. Update the credentials for your new client (in MyRealm-realm.json)
    a. In Keycloak -> Manage Realms -> MyRealm -> Client -> <My backend client>
    b. Click on "Credentials" tabkeycloak
    c. Copy the credentials
    d. Open this file:  infrastructure/kubernetes/MyRealm-realm.json
    e. Search for
            the credentials (you copied earlier)
       Replace the credentials with
            client-secret-set-in-keycloak

  4. The keycloak settings work

        cd %NI2_PROJECT_PATH%\infrastructure\kubernetes\

        kubectl create configmap realm-export   --from-file=MyRealm-realm.json

    b. Restart the keycloak deployment
       cd %NI2_PROJECT_PATH%\infrastructure\kubernetes
       kubectl delete -f deploy-keycloak.yaml

    c. Connect to keycloak
       Go to https://keycloak.docker.internal/

    d. Verify that you see your changes
       1) Connect to https://keycloak.docker.internal
       2) Login as admin
          Password is admin
       3) Go to Manage Realms -> MyRealm -> Clients

  5. Verify that your KEYCLOAK_CLIENT_ID matches your real client id in keycloak
